63 inline void ComplexWriteFwdInterleavedT1(__m512d arg_xr, __m512d arg_yr,
64  __m512d arg_xi, __m512d arg_yi,
65  __m512d* out) {
66  const __m512i vperm_4hi_4lo_idx = _mm512_set_epi64(3, 2, 1, 0, 7, 6, 5, 4);
67  const __m512i v_X_out_idx = _mm512_set_epi64(3, 1, 7, 5, 2, 0, 6, 4);
68  const __m512i v_Y_out_idx = _mm512_set_epi64(7, 5, 3, 1, 6, 4, 2, 0);
69 
70  // Real part
71  // in: 14r, 12r, 10r, 8r, 6r, 4r, 2r, 0r
72  // -> 6r, 4r, 2r, 0r, 14r, 12r, 10r, 8r
73  arg_xr = _mm512_permutexvar_pd(vperm_4hi_4lo_idx, arg_xr);
74 
75  // arg_yr: 15r, 13r, 11r, 9r, 7r, 5r, 3r, 1r
76  // -> 6r, 4r, 2r, 0r, 7r, 5r, 3r, 1r
77  __m512d perm_1 = _mm512_mask_blend_pd(0x0f, arg_xr, arg_yr);
78  // -> 15r, 13r, 11r, 9r, 14r, 12r, 10r, 8r
79  __m512d perm_2 = _mm512_mask_blend_pd(0xf0, arg_xr, arg_yr);
80 
81  // 7r, 3r, 6r, 2r, 5r, 1r, 4r, 0r
82  arg_xr = _mm512_permutexvar_pd(v_X_out_idx, perm_1);
83  // 15r, 11r, 14r, 10r, 13r, 9r, 12r, 8r
84  arg_yr = _mm512_permutexvar_pd(v_Y_out_idx, perm_2);
85 
86  // Imaginary part
87  // in: 14i, 12i, 10i, 8i, 6i, 4i, 2i, 0i
88  // -> 6i, 4i, 2i, 0i, 14i, 12i, 10i, 8i
89  arg_xi = _mm512_permutexvar_pd(vperm_4hi_4lo_idx, arg_xi);
90 
91  // arg_yr: 15i, 13i, 11i, 9i, 7i, 5i, 3i, 1i
92  // -> 6i, 4i, 2i, 0i, 7i, 5i, 3i, 1i
93  perm_1 = _mm512_mask_blend_pd(0x0f, arg_xi, arg_yi);
94  // -> 15i, 13i, 11i, 9i, 14i, 12i, 10i, 8i
95  perm_2 = _mm512_mask_blend_pd(0xf0, arg_xi, arg_yi);
96 
97  // 7i, 3i, 6i, 2i, 5i, 1i, 4i, 0i
98  arg_xi = _mm512_permutexvar_pd(v_X_out_idx, perm_1);
99  // 15i, 11i, 14i, 10i, 13i, 9i, 12i, 8i
100  arg_yi = _mm512_permutexvar_pd(v_Y_out_idx, perm_2);
101 
102  // Merge
103  // 00000000 > 3i 3r 2i 2r 1i 1r 0i 0r
104  __m512d out1 = _mm512_shuffle_pd(arg_xr, arg_xi, 0x00);
105  // 11111111 > 7i 7r 6i 6r 5i 5r 4i 4r
106  __m512d out2 = _mm512_shuffle_pd(arg_xr, arg_xi, 0xff);
107 
108  // 00000000 > 11i 11r 10i 10r 9i 9r 8i 8r
109  __m512d out3 = _mm512_shuffle_pd(arg_yr, arg_yi, 0x00);
110  // 11111111 > 15i 15r 14i 14r 13i 13r 12i 12r
111  __m512d out4 = _mm512_shuffle_pd(arg_yr, arg_yi, 0xff);
112 
113  _mm512_storeu_pd(out++, out1);
114  _mm512_storeu_pd(out++, out2);
115  _mm512_storeu_pd(out++, out3);
116  _mm512_storeu_pd(out++, out4);
117 }
